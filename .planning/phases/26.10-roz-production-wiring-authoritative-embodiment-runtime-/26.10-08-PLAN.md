       //! FW-07 — Deterministic fake manipulator backend.
       //!
       //! Mirrors `roz_mavlink::backend::MavlinkBackend`'s shared-state pattern: actuator
       //! and sensor halves share `Arc<Mutex<FakeManipulatorState>>` so commands flow into
       //! the next sensor read.
       //!
       //! Lives in roz-copper (not roz-test) to avoid a cargo cycle: roz-test already
       //! depends on roz-copper.

       #![cfg(any(test, feature = "test-fixtures"))]

       use std::sync::{Arc, Mutex};
       use std::sync::atomic::{AtomicI64, Ordering};

       use crate::io::{ActuatorSink, SensorFrame, SensorSource};
       use roz_core::command::CommandFrame;
       use roz_core::embodiment::EmbodimentRuntime;

       /// Tick period for the integrator. Mirrors Copper's 100 Hz tick.
       pub const FAKE_TICK_PERIOD_NS: i64 = 10_000_000;

       /// Inner state shared between the actuator and sensor halves.
       struct FakeManipulatorState {
           joint_positions: Vec<f64>,
           joint_commands: Vec<f64>,
           joint_max_velocity: Vec<f64>,
           joint_position_bounds: Vec<(f64, f64)>,
           sim_time_ns: AtomicI64,
           command_count: usize,
       }

       /// Actuator half of the shared backend. Cheap to clone (Arc).
       #[derive(Clone)]
       pub struct FakeManipulatorActuator(Arc<Mutex<FakeManipulatorState>>);

       /// Sensor half of the shared backend.
       pub struct FakeManipulatorSensor(Arc<Mutex<FakeManipulatorState>>);

       /// Construct a paired fake backend keyed off the runtime's joint set.
       pub fn fake_manipulator_pair(runtime: &EmbodimentRuntime) -> (FakeManipulatorActuator, FakeManipulatorSensor) {
           let n = runtime.model.joints.len();
           let state = FakeManipulatorState {
               joint_positions: vec![0.0; n],
               joint_commands: vec![0.0; n],
               joint_max_velocity: runtime.model.joints.iter()
                   .map(|j| j.limits.max_velocity)
                   .collect(),
               joint_position_bounds: runtime.model.joints.iter()
                   .map(|j| (j.limits.position_min, j.limits.position_max))
                   .collect(),
               sim_time_ns: AtomicI64::new(0),
               command_count: n,
           };
           let shared = Arc::new(Mutex::new(state));
           (FakeManipulatorActuator(Arc::clone(&shared)), FakeManipulatorSensor(shared))
       }

       impl ActuatorSink for FakeManipulatorActuator {
           fn send(&self, frame: &CommandFrame) -> anyhow::Result<()> {
               let mut state = self.0.lock().expect("fake-manipulator mutex poisoned");
               for (i, &v) in frame.values.iter().enumerate().take(state.command_count) {
                   let max = state.joint_max_velocity[i];
                   state.joint_commands[i] = v.clamp(-max, max);
               }
               Ok(())
           }
       }

       impl SensorSource for FakeManipulatorSensor {
           fn try_recv(&mut self) -> Option<SensorFrame> {
               let mut state = self.0.lock().expect("fake-manipulator mutex poisoned");
               let now = state.sim_time_ns.fetch_add(FAKE_TICK_PERIOD_NS, Ordering::Relaxed) + FAKE_TICK_PERIOD_NS;
               let dt = (FAKE_TICK_PERIOD_NS as f64) / 1e9;

               // Integrate commanded velocity into position; clamp to bounds.
               let cmds = state.joint_commands.clone();
               for (i, v) in cmds.iter().enumerate() {
