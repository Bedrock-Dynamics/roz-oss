```rust
// Load embodiment model from robot.toml if configured (D-01, D-02)
let embodiment_model: Option<roz_core::embodiment::model::EmbodimentModel> =
    if let Some(ref toml_path) = config.robot_toml {
        match roz_core::manifest::EmbodimentManifest::load(std::path::Path::new(toml_path)) {
            Ok(manifest) => match manifest.embodiment_runtime() {
                Some(rt) => {
                    tracing::info!(path = %toml_path, digest = %rt.model.model_digest, "loaded embodiment model from manifest");
                    Some(rt.model)
                }
                None => {
                    tracing::info!(path = %toml_path, "manifest has no channels section, skipping embodiment upload");
                    None
                }
            },
            Err(e) => {
                tracing::warn!(path = %toml_path, error = %e, "failed to load embodiment manifest");
                None
            }
        }
    } else {
        None
    };
```

```rust
    tracing::info!(host_id = %host_id, "registered with server");
    // Upload embodiment model if available (D-04: log-and-continue on failure, D-05: None for runtime)
    if let Some(ref model) = embodiment_model {

    }
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| filesystem->worker | Worker reads robot.toml from operator-configured path |
| worker->server REST API | Worker sends serialized EmbodimentModel over authenticated PUT |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-05 | Information Disclosure | robot_toml path | accept | Path is operator-configured via env var or TOML config, not user-facing input. Operator controls file access. |
| T-05-06 | Tampering | manifest file content | accept | Worker reads a local file configured by the operator. If the operator's filesystem is compromised, the worker is already compromised. |
| T-05-07 | Denial of Service | large manifest file | accept | Manifest files are small TOML (<10KB). `EmbodimentManifest::load` uses `std::fs::read_to_string` which will fail on truly huge files. One-time startup read. |
| T-05-08 | Spoofing | upload_embodiment auth | mitigate | Uses existing bearer auth (`config.api_key`) already validated on server side by Extension(auth) middleware |
</threat_model>
