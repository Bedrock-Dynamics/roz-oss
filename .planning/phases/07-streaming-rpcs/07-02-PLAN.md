**5. Add `compute_frame_tree_delta` helper function** (private, after the `EmbodimentService` impl block):

```rust
/// Compute the delta between two FrameTrees.
///
/// Returns a `FrameTreeDelta` containing changed/added frames, removed frame IDs,
/// and an optional new root if the root changed.
fn compute_frame_tree_delta(
    old: &roz_core::embodiment::frame_tree::FrameTree,
    new: &roz_core::embodiment::frame_tree::FrameTree,
) -> FrameTreeDelta {
    let old_ids: std::collections::HashSet<&str> = old.all_frame_ids().into_iter().collect();
    let new_ids: std::collections::HashSet<&str> = new.all_frame_ids().into_iter().collect();

    let mut changed_frames = BTreeMap::new();

    // Find changed and added frames
    for id in &new_ids {
        let new_node = new.get_frame(id).expect("frame exists in new tree");
        match old.get_frame(id) {
            Some(old_node) if old_node == new_node => {} // unchanged
            _ => {
                // Changed or added
                changed_frames.insert(id.to_string(), crate::grpc::roz_v1::FrameNode::from(new_node));
            }
        }
    }

    // Find removed frames
    let removed_frame_ids: Vec<String> = old_ids
        .difference(&new_ids)
        .map(|id| id.to_string())
        .collect();

    // Check if root changed
    let new_root = if old.root() != new.root() {
        new.root().map(String::from)
    } else {
        None
    };

    FrameTreeDelta {
        changed_frames,
        removed_frame_ids,
        new_root,
    }
}
```

```rust
/// Compute the delta between two CalibrationOverlays.
///
/// Compares BTreeMap fields entry-by-entry and reports changed/removed entries.
/// Also includes scalar metadata (calibration_id, calibrated_at) if they changed.
fn compute_calibration_delta(
    old: &roz_core::embodiment::calibration::CalibrationOverlay,
    new: &roz_core::embodiment::calibration::CalibrationOverlay,
) -> CalibrationDelta {
    // Joint offsets
    let mut changed_joint_offsets = BTreeMap::new();
    let mut removed_joint_offsets = Vec::new();
    for (key, &val) in &new.joint_offsets {
        match old.joint_offsets.get(key) {
            Some(&old_val) if (old_val - val).abs() < f64::EPSILON => {} // unchanged
            _ => { changed_joint_offsets.insert(key.clone(), val); }
        }
    }
    for key in old.joint_offsets.keys() {
        if !new.joint_offsets.contains_key(key) {
            removed_joint_offsets.push(key.clone());
        }
    }

    // Frame corrections
    let mut changed_frame_corrections = BTreeMap::new();
    let mut removed_frame_corrections = Vec::new();
    for (key, val) in &new.frame_corrections {
        match old.frame_corrections.get(key) {
            Some(old_val) if old_val == val => {} // unchanged
            _ => { changed_frame_corrections.insert(key.clone(), crate::grpc::roz_v1::Transform3D::from(val)); }
        }
    }
    for key in old.frame_corrections.keys() {
        if !new.frame_corrections.contains_key(key) {
            removed_frame_corrections.push(key.clone());
        }
    }

    // Sensor calibrations
    let mut changed_sensor_calibrations = BTreeMap::new();
    let mut removed_sensor_calibrations = Vec::new();
    for (key, val) in &new.sensor_calibrations {
        match old.sensor_calibrations.get(key) {
            Some(old_val) if old_val == val => {} // unchanged
            _ => { changed_sensor_calibrations.insert(key.clone(), crate::grpc::roz_v1::SensorCalibration::from(val)); }
        }
    }
    for key in old.sensor_calibrations.keys() {
        if !new.sensor_calibrations.contains_key(key) {
            removed_sensor_calibrations.push(key.clone());
        }
    }

    // Scalar metadata (per research Open Question 2 RESOLVED)
    let calibration_id = if old.calibration_id != new.calibration_id {
        Some(new.calibration_id.clone())
    } else {
        None
    };
    let calibrated_at = if old.calibrated_at != new.calibrated_at {
        Some(prost_types::Timestamp::from(std::time::SystemTime::from(new.calibrated_at)))
    } else {
        None
    };

    CalibrationDelta {
        changed_joint_offsets,
        removed_joint_offsets,
        changed_frame_corrections,
        removed_frame_corrections,
        changed_sensor_calibrations,
        removed_sensor_calibrations,
        calibration_id,
        calibrated_at,
    }
}
```
