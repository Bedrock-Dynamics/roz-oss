/// Drain report produced by `await_turn_drain` (REVIEWS.md M3).
#[derive(Debug, Default)]
struct DrainReport {
    saw_turn_finished: bool,
    saw_model_call: bool,
    saw_tool_call_requested: bool,
    saw_tool_call_started: bool,
    saw_tool_call_finished: bool,

/// REVIEWS.md M3: deterministic drain barrier.
///
/// Subscribe a SECOND receiver to the event broadcast BEFORE starting the
/// turn; then after `run_turn_streaming` returns, call this helper to poll
/// the second receiver with a timeout until `TurnFinished` arrives AND all
/// expected gap events (TurnStarted, ModelCallCompleted, ToolCallRequested,
/// ToolCallStarted, ToolCallFinished) have been observed. This guarantees
/// the cloud ingestor (which is also subscribed) has had a chance to enqueue
/// every event into the MCAP writer before `Finalize` is sent.
///
/// Replaces the flaky `tokio::time::sleep(200ms)` drain used in earlier
/// plan drafts.
async fn await_turn_drain<Rx>(mut rx: Rx, timeout: Duration) -> DrainReport
where
    // Adjust the trait bound to match the runtime's actual receiver type.
    // If the receiver is `tokio::sync::broadcast::Receiver<EventEnvelope>`,
    // bound to that. If it's a mpsc::Receiver or a custom type, bound to
    // that. The body below assumes a method like `recv().await -> Option<T>`
    // or `recv().await -> Result<T, RecvError>` — adapt.
    Rx: /* whatever the runtime's subscribe_events returns */,
{
    let mut report = DrainReport::default();
    let deadline = tokio::time::Instant::now() + timeout;

    loop {
        let now = tokio::time::Instant::now();
        if now >= deadline {
            break;
        }
        let remaining = deadline - now;

        let maybe = tokio::time::timeout(remaining, rx.recv()).await;
        let envelope = match maybe {
            Ok(Ok(env)) => env,
            Ok(Err(_recv_err)) => break,   // channel closed or lagged; stop
            Err(_elapsed) => break,         // timeout
        };

        report.total_events += 1;

        // Extract the event_type string from the envelope. The exact
        // accessor depends on the type — adjust per the runtime's envelope
        // shape. Pattern-match on the SessionEvent variant OR read a
        // canonical_event_type_name string.
        let ety = /* extract event_type string from envelope */;

        match ety.as_str() {
            "turn_started" => report.saw_turn_started = true,
            "model_call" => report.saw_model_call = true,
            "tool_call_requested" => report.saw_tool_call_requested = true,
            "tool_call_started" => report.saw_tool_call_started = true,
            "tool_call_finished" => report.saw_tool_call_finished = true,
            "turn_finished" => {
                report.saw_turn_finished = true;
                // TurnFinished is terminal — we can break as soon as all
                // required gap events have also been seen.
                if report.saw_turn_started
                    && report.saw_model_call
                    && report.saw_tool_call_requested
                    && report.saw_tool_call_started
                    && report.saw_tool_call_finished
                {
                    break;
                }
            }
            _ => {}
        }
    }

    // Give the cloud ingestor's background task a chance to enqueue the
    // WriteCommand::Event items it has collected, since each enqueue is a
    // separate .await point inside the ingestor.
    for _ in 0..8 {
        tokio::task::yield_now().await;
    }

    report
}
